function calcular(){

let distancia=document.getElementById('distancia').value
let vehiculo=document.getElementById('vehiculo').value
let condicion=document.getElementById('condicion').value
let horario=document.getElementById('horario').value
let via=document.getElementById('via').value

let base=500
let total=base+(distancia*18)

total=total*vehiculo*condicion*horario*via

document.getElementById('resultado').innerHTML="Cotización estimada: $"+total.toFixed(2)+" MXN"

window.totalCotizacion=total.toFixed(2)

}

function generarPDF(){

const {jsPDF}=window.jspdf
let doc=new jsPDF()

let precio=window.totalCotizacion || "No calculado"

doc.text("COTIZACIÓN CALCUGRUS",20,20)
doc.text("Precio estimado: $"+precio+" MXN",20,40)
doc.text("Cotización generada automáticamente",20,60)

doc.save("cotizacion_calcugrus.pdf")

}
